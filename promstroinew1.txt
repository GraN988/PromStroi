    const galleryItems = document.querySelectorAll('.gallery-item');
    const lightbox = document.getElementById('lightbox');
    const lightboxImage = document.getElementById('lightbox-image');
    const lightboxCaption = document.getElementById('lightbox-caption');
    const lightboxClose = document.getElementById('lightbox-close');
    const lightboxPrev = document.getElementById('lightbox-prev');
    const lightboxNext = document.getElementById('lightbox-next');

    let currentType = null;
    let currentIndex = 0;

    function openLightbox(type) {
      currentType = type;
      currentIndex = 0;
      updateLightbox();
      lightbox.classList.add('active');
      lightbox.focus();
    }

    function closeLightbox() {
      lightbox.classList.remove('active');
      currentType = null;
      currentIndex = 0;
    }

    function updateLightbox() {
      const images = galleryData[currentType];
      if (!images) return;
      const image = images[currentIndex];
      lightboxImage.src = image.src;
      lightboxImage.alt = image.alt;
      lightboxCaption.textContent = image.caption;
    }

    function showPrev() {
      const images = galleryData[currentType];
      if (!images) return;
      currentIndex = (currentIndex - 1 + images.length) % images.length;
      updateLightbox();
    }

    function showNext() {
      const images = galleryData[currentType];
      if (!images) return;
      currentIndex = (currentIndex + 1) % images.length;
      updateLightbox();
    }

    galleryItems.forEach(item => {
      item.addEventListener('click', () => {
        const type = item.getAttribute('data-type');
        openLightbox(type);
      });
      item.addEventListener('keydown', e => {
        if (e.key === 'Enter' || e.key === ' ') {
          e.preventDefault();
          const type = item.getAttribute('data-type');
          openLightbox(type);
        }
      });
    });

    lightboxClose.addEventListener('click', closeLightbox);
    lightboxPrev.addEventListener('click', showPrev);
    lightboxNext.addEventListener('click', showNext);

    // Закрытие по клику вне картинки
    lightbox.addEventListener('click', e => {
      if (e.target === lightbox) closeLightbox();
    });

    // Клавиши навигации в лайтбоксе
    document.addEventListener('keydown', e => {
      if (!lightbox.classList.contains('active')) return;
      if (e.key === 'Escape') {
        closeLightbox();
      } else if (e.key === 'ArrowLeft') {
        showPrev();
      } else if (e.key === 'ArrowRight') {
        showNext();
      }
    });

    // Обработка формы с Formspree
    const form = document.getElementById('contactForm');
    const formMessage = document.getElementById('form-message');

    form.addEventListener('submit', async (e) => {
      e.preventDefault();

      const formData = new FormData(form);

      // Простая валидация телефона (можно улучшить)
      const phone = formData.get('phone').replace(/\D/g, '');
      if (phone.length < 6) {
        alert('Пожалуйста, введите корректный номер телефона.');
        return;
      }

      try {
        const response = await fetch(form.action, {
          method: form.method,
          headers: { 'Accept': 'application/json' },
          body: formData
        });

        if (response.ok) {
          formMessage.style.color = 'green';
          formMessage.textContent = 'Спасибо! Ваше сообщение отправлено.';
          form.reset();
        } else {
          const data = await response.json();
          if (data.errors) {
            formMessage.style.color = 'red';
            formMessage.textContent = data.errors.map(error => error.message).join(', ');
          } else {
            formMessage.style.color = 'red';
            formMessage.textContent = 'Ошибка отправки. Попробуйте позже.';
          }
        }
      } catch (error) {
        formMessage.style.color = 'red';
        formMessage.textContent = 'Ошибка сети. Попробуйте позже.';
      }
    });
